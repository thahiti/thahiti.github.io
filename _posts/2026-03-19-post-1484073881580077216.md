import logging
import functools
import time
from typing import Any, Callable

# 1. 로거 설정
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(name)s - %(levelname)s - %(message)s')
logger = logging.getLogger("LangGraphNode")

def log_node(func: Callable):
    """LangGraph 노드의 시작과 끝을 로깅하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(state: Any, *args, **kwargs):
        node_name = func.__name__

        # 시작 로깅
        logger.info(f"--- [START] Node: {node_name} ---")
        logger.debug(f"Input State: {state}")

        start_time = time.perf_counter()

        try:
            # 실제 노드 함수 실행
            result = func(state, *args, **kwargs)

            end_time = time.perf_counter()
            duration = end_time - start_time

            # 종료 로깅
            logger.info(f"--- [END] Node: {node_name} (Duration: {duration:.4f}s) ---")
            logger.debug(f"Output Updates: {result}")

            return result

        except Exception as e:
            logger.error(f"--- [ERROR] Node: {node_name} ---")
            logger.exception(e)
            raise e

    return wrapper
