llm = ChatOpenAI(model="gpt-4o-mini")

@tool
def calc_tool(query: str) -> str:
    try:
        return str(eval(query))
    except Exception as e:
        return str(e)

@tool
def search_tool(query: str) -> str:
    return f"[{query}]에 대한 가상 검색 결과: 2026년 최신 정보입니다."

@tool
def transfer_to_calc(query: str):
    pass

@tool
def transfer_to_search(query: str):
    pass

def supervisor_node(state: MessagesState):
    bound_llm = llm.bind_tools([transfer_to_calc, transfer_to_search])
    response = bound_llm.invoke(state["messages"])

    if not response.tool_calls:
        return Command(update={"messages": [response]}, goto=END)

    t_call = response.tool_calls[0]
    goto = "calc_node" if t_call["name"] == "transfer_to_calc" else "search_node"
    return Command(update={"messages": [response]}, goto=goto)

def calc_node(state: MessagesState):
    last_msg = state["messages"][-1]
    t_call = last_msg.tool_calls[0]
    result = calc_tool.invoke(t_call["args"])
    msg = ToolMessage(content=str(result), tool_call_id=t_call["id"], name=t_call["name"])
    return Command(update={"messages": [msg]}, goto="supervisor")

def search_node(state: MessagesState):
    last_msg = state["messages"][-1]
    t_call = last_msg.tool_calls[0]
    result = search_tool.invoke(t_call["args"])
    msg = ToolMessage(content=str(result), tool_call_id=t_call["id"], name=t_call["name"])
    return Command(update={"messages": [msg]}, goto="supervisor")

builder = StateGraph(MessagesState)
builder.add_node("supervisor", supervisor_node)
builder.add_node("calc_node", calc_node)
builder.add_node("search_node", search_node)
builder.add_edge(START, "supervisor")

graph = builder.compile()

if __name__ == "__main__":
    inputs = {
        "messages": [
            HumanMessage(content="15 * 24를 계산해주고, 그 다음에 LangGraph 핸드오프에 대해 검색해줘.")
        ]
    }

    for event in graph.stream(inputs, stream_mode="values"):
        event["messages"][-1].pretty_print()
